1.1 for문 이용해서 테이블 출력하기

let table = "<table>";

    for( let i = 0; i < 10; i++ ){
        table += "<tr>";

        for( let j = 0; j < 10; j++){            // for문 안의 for문
            table += "<td>😎</td>"
        }
        table += "</tr>";
    }
    table += "</table>";

document.write(table);
